#include <Servo.h> //Biblioteka odpowiedzialna za serwa

 
Servo servoRFH;  //Tworzymy obiekt, dzięki któremu możemy odwołać się do serwa  //Aktualna pozycja serwa 0-180

void setup() 
{ 
  Serial.begin(9600);
  // R-right, L-left, F-front, M-middle, B-back, H-horizontal, V-vertical
  servoRFH.attach(13);  
  servoRFV.attach(12);
  servoRMH.attach(11);  
  servoRMV.attach(10);
  servoRBH.attach(9);
  servoRBV.attach(8);
  servoLFH.attach(7);  
  servoLFV.attach(6);
  servoLMH.attach(5);  
  servoLMV.attach(4);
  servoLBH.attach(3);
  servoLBV.attach(2);
  //nogi RF, LM i RB działają synchronicznie, jak również nogi LF, RM i LB
} 
 
void loop() 
{  
  if(Serial.available() > 0){

    char incoming = Serial.read();

    if (isAlpha(incoming)) {   // tylko litery
      tryb = incoming;
    }
    if(tryb != oldtryb){ //powrót do pozycji wyjściowej po zmianie trybu
      oldtryb=tryb;
      servoRFH.write(mid);
      delay(50);
      servoRMH.write(mid);
      delay(50);
      servoRBH.write(mid);
      delay(50);
      servoLFH.write(mid);
      delay(50);
      servoLMH.write(mid);
      delay(50);
      servoLBH.write(mid);
      delay(100);
    }
  }
  Serial.println(tryb);
  executeCommand(tryb);
  delay(10);
}
